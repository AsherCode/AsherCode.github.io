部署

  • 拷贝blog-search elasticsearch2.4.4 mongodb-file-server-master,分配AsherCode权限chmod 777 elasticsearch2.4.4 -R,nohup ./elasticsearch &,下载mongodb-linux-x86_64-rhel70-3.2.4.gz
  • nohup ./mongod –dbpath /root/mongodb3.2.4/data/db –logpath /root/mongodb3.2.4/logs/mongodb.log –bind_ip 0.0.0.0 & 启动mongodb

  • gradle build && nohup java -jar build/libs/mongodb-file-server-master-1.0.0.jar &

  • wget -O /etc/yum.repos.d/CentOS-Base.repo http://mirrors.aliyun.com/repo/Centos-7.repo && yum clean all && yum makecache修改yum源

  • wget http://repo.mysql.com/mysql-community-release-el7-5.noarch.rpm && rpm -ivh mysql-community-release-el7-5.noarch.rpm
  • yum install mysql-server && mysql -u root
    • use mysql; update user set authentication_string=password(‘123456’) where user=’root’; flush privileges;
    • CREATE DATABASE db_name DEFAULT CHARACTER SET utf8 COLLATE utf8_general_ci;
  • cd blog-search && gradle build && java -jar build/libs/blog-search-1.0.0.jar
